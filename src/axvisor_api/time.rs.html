1//! Time and timer APIs.
2
3extern crate alloc;
4
5use alloc::boxed::Box;
6use core::time::Duration;
7
8/// Time value.
9pub type TimeValue = Duration;
10/// Nanoseconds count.
11pub type Nanos = u64;
12/// Tick count.
13pub type Ticks = u64;
14/// Cancel token, used to cancel a scheduled timer event.
15pub type CancelToken = usize;
16
17/// The API trait for time and timer functionalities.
18#[crate::api_def]
19pub trait TimeIf {
20    /// Get the current tick count.
21    fn current_ticks() -> Ticks;
22    /// Convert ticks to nanoseconds.
23    fn ticks_to_nanos(ticks: Ticks) -> Nanos;
24    /// Convert nanoseconds to ticks.
25    fn nanos_to_ticks(nanos: Nanos) -> Ticks;
26    /// Register a timer.
27    fn register_timer(
28        deadline: TimeValue,
29        callback: Box<dyn FnOnce(TimeValue) + Send + 'static>,
30    ) -> CancelToken;
31    /// Cancel a timer.
32    fn cancel_timer(token: CancelToken);
33}
34
35/// Get the current time in nanoseconds.
36pub fn current_time_nanos() -> Nanos {
37    ticks_to_nanos(current_ticks())
38}
39
40/// Get the current time.
41pub fn current_time() -> TimeValue {
42    Duration::from_nanos(current_time_nanos())
43}
44
45/// Convert ticks to time.
46pub fn ticks_to_time(ticks: Ticks) -> TimeValue {
47    Duration::from_nanos(ticks_to_nanos(ticks))
48}
49
50/// Convert time to ticks.
51pub fn time_to_ticks(time: TimeValue) -> Ticks {
52    nanos_to_ticks(time.as_nanos() as Nanos)
53}
